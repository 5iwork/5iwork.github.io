gitlab集成crowd

/etc/gitlab/gitlab.rb
gitlab_rails['omniauth_enabled'] = true
gitlab_rails['omniauth_allow_single_sign_on'] = ['crowd']
gitlab_rails['omniauth_block_auto_created_users'] = true
gitlab_rails['omniauth_providers'] = [
    {
      "name" => "crowd",
      "label" => "Crowd域账号", # optional label for login button, defaults to "Crowd"
      "args" => {
        "crowd_server_url" => "CROWD_SERVER_URL",
        "application_name" => "YOUR_APP_NAME",
        "application_password" => "YOUR_APP_PASSWORD"
      }
    }
]
详见：https://docs.gitlab.com/ee/administration/auth/crowd.html#configure-a-new-crowd-application

  1. 重新编译 sudo gitlab-ctl reconfigure 并重启 sudo gitlab-ctl restart
